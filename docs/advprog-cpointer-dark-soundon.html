課堂練習：C Pointers

課堂練習：Pointer Arithmetic:

Example 1: Incrementing and decrementing pointers:

#include <stdio.h>

int main() {
    int arr[] = {1, 2, 3, 4, 5};
    int *ptr = arr;

    printf("Original pointer: %p\\n", ptr);

    ptr++; // Increment pointer
    printf("After increment: %p\\n", ptr);

    ptr--; // Decrement pointer
    printf("After decrement: %p\\n", ptr);

    return 0;
}

Example 2: Adding an integer value to a pointer:

#include <stdio.h>

int main() {
    int arr[] = {1, 2, 3, 4, 5};
    int *ptr = arr;

    printf("Original pointer: %p\\n", ptr);

    ptr += 2; // Add 2 to the pointer
    printf("After adding 2: %p\\n", ptr);

    return 0;
}

課堂練習：Dynamic Memory Allocation:

Example 1: Allocating and freeing memory using malloc():

#include <stdio.h>
#include <stdlib.h>

int main() {
    int *ptr;

    ptr = (int *)malloc(5 * sizeof(int));
    if (ptr == NULL) {
        printf("Memory allocation failed!\\n");
        return 1;
    }

    // Use the allocated memory

    free(ptr);

    return 0;
}

Example 2: Reallocating memory using realloc():

#include <stdio.h>
#include <stdlib.h>

int main() {
    int *ptr;

    ptr = (int *)malloc(5 * sizeof(int));
    if (ptr == NULL) {
        printf("Memory allocation failed!\\n");
        return 1;
    }

    // Use the allocated memory

    ptr = (int *)realloc(ptr, 10 * sizeof(int));
    if (ptr == NULL) {
        printf("Memory reallocation failed!\\n");
        return 1;
    }

    // Use the reallocated memory

    free(ptr);

    return 0;
}

課堂練習：Pointers and Functions:

Example 1: Passing a pointer to a function:

#include <stdio.h>

void increment(int *num) {
    (*num)++;
}

int main() {
    int x = 5;

    printf("Before increment: %d\\n", x);
    increment(&x);
    printf("After increment: %d\\n", x);

    return 0;
}

Example 2: Returning a pointer from a function:

#include <stdio.h>
#include <stdlib.h>

int *createArray(int size) {
    int *arr = (int *)malloc(size * sizeof(int));
    // Populate the array
    return arr;
}

int main() {
    int *ptr;
    int size = 5;

    ptr = createArray(size);
    if (ptr == NULL) {
        printf("Memory allocation failed!\\n");
        return 1;
    }

    // Use the array

    free(ptr);

    return 0;
}

課堂練習：Pointers and Arrays:

Example 1: Using pointer arithmetic to access array elements:

#include <stdio.h>

int main() {
    int arr[] = {1, 2, 3, 4, 5};
    int *ptr = arr;

    printf("First element: %d\\n", *ptr);
    printf("Third element: %d\\n", *(ptr + 2));

    return 0;
}

Example 2: Creating an array of pointers:

#include <stdio.h>

int main() {
    int x = 10, y = 20, z = 30;
    int *arr[3];

    arr[0] = &x;
    arr[1] = &y;
    arr[2] = &z;

    printf("First element: %d\\n", *arr[0]);
    printf("Third element: %d\\n", *arr[2]);

    return 0;
}

課堂練習：Pointers to Pointers:

Example 1: Using double indirection to modify a value:

#include <stdio.h>

int main() {
    int x = 5;
    int *ptr = &x;
    int **ptrToPtr = &ptr;

    printf("Before modification: %d\\n", **ptrToPtr);

    **ptrToPtr = 10;

    printf("After modification: %d\\n", **ptrToPtr);

    return 0;
}

Example 2: Accessing a value through a pointer to a pointer:

#include <stdio.h>

int main() {
    int x = 5;
    int *ptr = &x;
    int **ptrToPtr = &ptr;

    printf("Value through double indirection: %d\\n", **ptrToPtr);

    return 0;
}

課堂練習：Pointers and Strings:

Example 1: Accessing and printing string characters using pointers:

#include <stdio.h>

int main() {
    char *str = "Hello";

    while (*str != '\\0') {
        printf("%c ", *str);
        str++;
    }

    return 0;
}

Example 2: Using string functions with pointers:

#include <stdio.h>
#include <string.h>

int main() {
    char str1[20] = "Hello";
    char str2[20] = "World";

    strcat(str1, str2);

    printf("Concatenated string: %s\\n", str1);

    return 0;
}
