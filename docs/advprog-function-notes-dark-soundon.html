Function Basics

// Function declaration
int addNumbers(int a, int b);

// Function definition
int addNumbers(int a, int b) {
    return a + b;
}
int multiply(int x, int y) {
    int result = x * y;
    return result;

int multiply(int x, int y);

int main() {
    int result = multiply(3, 5);
    printf("Result: %d\\n", result);

}

// Function definition
int multiply(int x, int y) {
    int result = x * y;
    return result;
}
int square(int num) {
    return num * num;
}

int main() {
    int result = square(5);  // Function call
    printf("Result: %d\\n", result);
    return 0;
}
int globalVar = 10;

void updateGlobal() {
    globalVar = 20;  // Accessing the global variable
}

int main() {
    int localVar = 5;  // Local variable
    printf("Local variable: %d\\n", localVar);
    printf("Global variable: %d\\n", globalVar);
    updateGlobal();
    printf("Updated global variable: %d\\n", globalVar);
    return 0;
}
void increment(int num) {
    num++;  // Changes the local copy, not affecting the original value
}

void incrementByAddress(int* numPtr) {
    (*numPtr)++;  // Modifies the value at the memory address
}

int main() {
    int value = 5;
    increment(value);
    printf("Value after increment: %d\\n", value);  // Output: 5

    incrementByAddress(&value);
    printf("Value after increment by address: %d\\n", value);  // Output: 6

    return 0;
}

User-Defined Functions

// Function declaration
int max(int a, int b);

// Function definition
int max(int a, int b) {
    return (a > b) ? a : b;
}

int main() {
    int result = max(3, 5);
    printf("Max value: %d\\n", result);
    return 0;
}
int square(int num) {  // Function header
    int result = num * num;  // Function body
    return result;
}
int square(int num) {
    return num * num;  // Return statement
}

int main() {
    int result = square(5);
    printf("Result: %d\\n", result);
    return 0;
}

int factorial(int n) {
    if (n == 0)
        return 1;  // Base case
    else
        return n * factorial(n - 1);  // Recursive case
}

int main() {
    int result = factorial(5);
    printf("Result: %d\\n", result);
    return 0;
}

#include <stdio.h>
#include <ctype.h>
#include <string.h>

int main() {
    char ch = 'A';
    if (isalpha(ch))
        printf("%c is an alphabet.\\n", ch);
    else
        printf("%c is not an alphabet.\\n", ch);

    char lowercase = tolower(ch);
    printf("Lowercase: %c\\n", lowercase);

    char str[] = "Hello";
    int length = strlen(str);
    printf("Length of the string: %d\\n", length);

    return 0;
}
#include <stdio.h>
#include <string.h>

int main() {
    char str1[20] = "Hello";
    char str2[] = " World";
    strcat(str1, str2);
    printf("Concatenated string: %s\\n", str1);

    char str3[] = "Apple";
    char str4[] = "Banana";
    int comparison = strcmp(str3, str4);
    if (comparison == 0)
        printf("Strings are equal.\\n");
    else if (comparison < 0)
        printf("%s comes before %s.\\n", str3, str4);
    else
        printf("%s comes after %s.\\n", str3, str4);

    char str5[] = "Hello, World!";
    char sub[] = "World";
    char* ptr = strstr(str5, sub);
    if (ptr != NULL)
        printf("Substring found at index: %ld\\n", ptr - str5);
    else
        printf("Substring not found.\\n");

    return 0;
}

#include <stdio.h>
#include <stdlib.h>

int main() {
    int* ptr = (int*)malloc(sizeof(int));
    if (ptr != NULL) {
        *ptr = 10;
        printf("Value: %d\\n", *ptr);
        free(ptr);
    }

    int* arr = (int*)calloc(5, sizeof(int));
    if (arr != NULL) {
        for (int i = 0; i < 5; i++)
            printf("Element %d: %d\\n", i, arr[i]);
        free(arr);
    }

    return 0;
}
