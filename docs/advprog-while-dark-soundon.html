課堂練習：C while 迴圈練習

課堂練習：使用while迴圈計算1到100的總和並輸出結果。

#include <stdio.h>

int main() {
    int num = 1;
    int sum = 0;

    while (num <= 100) {
        sum += num;
        num++;
    }

    printf("Sum: %d\\n", sum);

    return 0;
}

課堂練習：使用do...while迴圈計算1到10的總和並輸出結果。

#include <stdio.h>

int main() {
    int num = 1;
    int sum = 0;

    do {
        sum += num;
        num++;
    } while (num <= 10);

    printf("Sum: %d\\n", sum);

    return 0;
}

課堂練習：使用do...while迴圈輸出10到1的數字。

#include <stdio.h>

int main() {
    int num = 10;

    do {
        printf("%d ", num);
        num--;
    } while (num >= 1);

    printf("\\n");

    return 0;
}

課堂練習：使用while迴圈計算1到n的階乘並輸出結果，其中n是使用者輸入的數字。

#include <stdio.h>

int main() {
    int n;
    int num = 1;
    int factorial = 1;

    printf("Enter a number: ");
    scanf("%d", &n);

    while (num <= n) {
        factorial *= num;
        num++;
    }

    printf("%d! = %d\\n", n, factorial);

    return 0;
}
