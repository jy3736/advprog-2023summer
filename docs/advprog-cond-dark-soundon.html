課堂練習：判斷輸入的數字是否為正數。如果是正數，則輸出「是正數」；否則，輸出「不是正數」。

#include <stdio.h>

int main() {
    int num;
    printf("請輸入一個數字：");
    scanf("%d", &num);

    if (num > 0) {
        printf("是正數\n");
    } else {
        printf("不是正數\n");
    }

    return 0;
}

課堂練習：判斷輸入的數字是否為質數。如果是質數，則輸出「是質數」；否則，輸出「不是質數」。

#include <stdio.h>
#include <stdbool.h>

bool isPrime(int num) {
    if (num <= 1) {
        return false;
    }

    for (int i = 2; i * i <= num; i++) {
        if (num % i == 0) {
            return false;
        }
    }

    return true;
}

int main() {
    int num;
    printf("請輸入一個數字：");
    scanf("%d", &num);

    if (isPrime(num)) {
        printf("是質數\\n");
    } else {
        printf("不是質數\\n");
    }

    return 0;
}

課堂練習：計算輸入的年份是否為潤年的第幾天（範圍為1-366）。如果是潤年，輸出對應的天數；否則，輸出「不是潤年」。

#include <stdio.h>

int isLeapYear(int year) {
    if ((year % 4 == 0 && year % 100 != 0) || year % 400 == 0) {
        return 1;
    } else {
        return 0;
    }
}

int getDayOfYear(int year, int month, int day) {
    int daysInMonth[] = {31, 28, 31, 30, 31, 30

, 31, 31, 30, 31, 30, 31};
    int totalDays = day;

    for (int i = 0; i < month - 1; i++) {
        totalDays += daysInMonth[i];
    }

    if (month > 2 && isLeapYear(year)) {
        totalDays++;
    }

    return totalDays;
}

int main() {
    int year, month, day;
    printf("請輸入日期（年 月 日，用空格分隔）：");
    scanf("%d %d %d", &year, &month, &day);

    if (isLeapYear(year)) {
        printf("是潤年的第 %d 天\\n", getDayOfYear(year, month, day));
    } else {
        printf("不是潤年\\n");
    }

    return 0;
}
