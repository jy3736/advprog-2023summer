Lecture: Example of C Struct Usage

#include <stdio.h>

struct Student {
    char id[10];
    int grade[10];
    int avg;
    int mid;
		int rank;
};

typedef struct Student STD;
// Function to sort an array in descending order
void sortArray(int arr[], int size) {
    for (int i = 0; i < size - 1; i++) {
        for (int j = 0; j < size - i - 1; j++) {
            if (arr[j] < arr[j + 1]) {
                int temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }
}

void calAvg(struct Student* ss) {
    // Calculate the average of all grades (avg)
    int sum = 0;
    int numGrades = 10; 
    for (int i = 0; i < numGrades; i++) {
        sum += ss->grade[i];
    }
    ss->avg = sum / numGrades;

    // Calculate the average of the best 5 grades (mid)
    int numBestGrades = 5;
    int bestGradesCopy[10];
    for (int i = 0; i < numGrades; i++) {
        bestGradesCopy[i] = ss->grade[i];
    }

    // Sort the copied grades array to get the best grades
    sortArray(bestGradesCopy, numGrades);

    // Calculate the average of the best grades
    int sumBestGrades = 0;
    for (int i = 0; i < numBestGrades; i++) {
        sumBestGrades += bestGradesCopy[i];
    }
    ss->mid = sumBestGrades / numBestGrades;
}

void calAvgAll(STD ss[], int size) {
    for(int i=0; i<size; i++)
        calAvg(&ss[i]);
}
void calRanking(STD ss[], int size) {
    float* g = (float*)calloc(size,sizeof(float));
    
    for(int i=0; i<size; i++)
        g[i] = ss[i].avg*0.6+ss[i].mid*0.4;
    for(int i=0; i<size; i++) {
        ss[i].rank = 1;
        for(int j=0; j<size; j++)
            if (g[i]<g[j]) ss[i].rank++;
    }
    
    free(g);
}
void dumpSTD(STD s) {
    printf("%10s ",s.id);
    for(int i=0; i<10; i++)
        printf("%4d ",s.grade[i]);
    printf("%4d %4d %4d\n",s.avg,s.mid,s.rank);
}

void dumpAll(STD ss[], int size) {
    for(int i=0; i<size; i++)
        dumpSTD(ss[i]);
}
int main() {
    STD s[] = {
        {"a12301",{50,40,60,70,80,90,90,70,70,100},0,0},
        {"a12302",{70,80,80,70,80,90,90,90,90,100},0,0},
        {"a12303",{30,50,20,50,50,30,20,50,60,70},0,0},
        {"a12304",{60,60,60,70,70,70,80,80,80,80},0,0},
        {"a12305",{80,90,80,0,0,90,90,80,100,100},0,0},        
        {"a12306",{30,50,20,50,50,30,20,50,60,70},0,0},
        {"a12307",{50,40,60,70,80,90,90,70,70,100},0,0},
    };
    int size = sizeof(s)/sizeof(STD);
    
    dumpAll(s,size);
    calAvgAll(s,size);
    dumpAll(s,size);
    calRanking(s,size);
    dumpAll(s,size);    
    
    return 0;
}
